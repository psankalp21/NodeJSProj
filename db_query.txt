SELECT comments.contents, users.username 
FROM comments 
JOIN users on users.id=comments.user_id 
WHERE comments.photo_id=3;


SELECT AVG(count_comment) AS average_comment
FROM (
  SELECT COUNT(*) AS count_comment
  FROM comments
  GROUP BY photo_id
) QUERY;
